//breadth first search
//Queue
CheckBipartite(G)
{
set up a new queue Q;
start at a random node N, color it black, and put N in Q;
while (Q is not empty)
{
x = pop(Q);
for every neighbor n of x
{
if (n is not black)
{
color n white and put n in Q;
}
else if (n is black)
{
break;
G IS NOT BIPARTITE;
}
}
}
G IS BIPARTITE;
}


%ENDCODE%

//Depth First Search
//Data structure: Stack
DFS-visit(v)
{
color v grey;
for each node n adjacent to v
{
if(n is white)
{
DFS-visit(n);
}
else
{
FOUND A CYCLE;
}
}
color black;
}
CycleDetection(G)
{
color all vertices white;
for each vertex v
{
DFS-visit(v);
}
}
